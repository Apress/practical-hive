USE census;
CREATE TABLE census.personname (
  persid         int,
  firstname      string,
  lastname       string
)
CLUSTERED BY (persid) INTO 1 BUCKETS
STORED AS orc 
TBLPROPERTIES('transactional' = 'true');

CREATE TABLE census.address (
  persid         int,
  postname      string
)
CLUSTERED BY (persid) INTO 1 BUCKETS
STORED AS orc 
TBLPROPERTIES('transactional' = 'true');

SELECT personname.firstname, 
personname.lastname,
address.postname 
FROM
census.personname
JOIN
census.address
ON (personname.persid = address.persid);
